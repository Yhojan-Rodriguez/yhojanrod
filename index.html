// Formulario de contacto const form = document.getElementById('contactForm'); const response = document.getElementById('formResponse'); form.addEventListener('submit', function(e){ e.preventDefault(); const name = form.name.value; const email = form.email.value; const message = form.message.value; if(name && email && message){ response.textContent = `Gracias ${name}, tu mensaje ha sido enviado!`; form.reset(); } else { response.textContent = "Por favor completa todos los campos."; } });
